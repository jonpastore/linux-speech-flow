CRITICAL: Use `Gtk.Window` with `set_modal(True)` — NOT `Gtk.Dialog` which is deprecated since GTK 4.10. The existing codebase uses `gi.require_version("Gtk", "4.0")`.

The dialog shows a list of failed WAV filenames (extracted from path basename, e.g. "recording_20260219T143022_abc3.wav"), one Gtk.CheckButton per file, pre-checked by default. Two action buttons at the bottom: "Reprocess Selected" and "Reprocess All".

"Reprocess All" triggers a mode selection: open a second small Gtk.Window (modal over the dialog) asking the user to choose "Write all to file" (mode="file") or "Paste each into current window" (mode="paste"). On choice, both windows close and on_selected is called.

"Reprocess Selected" uses mode="paste" (submits each selected WAV to the pipeline in sequence) and calls on_selected with the checked paths.

```python
import gi
gi.require_version("Gtk", "4.0")
from gi.repository import Gtk
from pathlib import Path


class ReprocessDialog(Gtk.Window):
    """Modal window listing failed WAV recordings for reprocess selection.

    Callers must use Gtk.Window (not Gtk.Dialog) — Gtk.Dialog deprecated GTK 4.10.
    """

    def __init__(self, failed_wavs: list[str], on_selected, application=None, parent=None):
        """
        Args:
            failed_wavs: List of absolute paths to failed WAV files.
            on_selected: Callable(wav_paths: list[str], mode: str) — mode is "paste" or "file".
            application: Gtk.Application for window registration.
            parent: Parent window for transient relationship.
        """
        super().__init__(title="Reprocess Recordings", application=application)
        self.set_modal(True)
        self.set_default_size(440, 320)
        if parent:
            self.set_transient_for(parent)

        self._failed_wavs = failed_wavs
        self._on_selected = on_selected
        self._checks: list[tuple[Gtk.CheckButton, str]] = []

        outer = Gtk.Box(orientation=Gtk.Orientation.VERTICAL, spacing=0)
        self.set_child(outer)

        scroll = Gtk.ScrolledWindow()
        scroll.set_vexpand(True)
        scroll.set_policy(Gtk.PolicyType.NEVER, Gtk.PolicyType.AUTOMATIC)
        outer.append(scroll)

        list_box = Gtk.Box(orientation=Gtk.Orientation.VERTICAL, spacing=4)
        list_box.set_margin_start(16)
        list_box.set_margin_end(16)
        list_box.set_margin_top(16)
        list_box.set_margin_bottom(8)
        scroll.set_child(list_box)

        header = Gtk.Label(label="Select recordings to reprocess:")
        header.set_xalign(0)
        header.add_css_class("title-4")
        list_box.append(header)

        for wav_path in failed_wavs:
            label = Path(wav_path).name
            check = Gtk.CheckButton(label=label)
            check.set_active(True)
            list_box.append(check)
            self._checks.append((check, wav_path))

        btn_row = Gtk.Box(orientation=Gtk.Orientation.HORIZONTAL, spacing=8)
        btn_row.set_margin_start(16)
        btn_row.set_margin_end(16)
        btn_row.set_margin_top(8)
        btn_row.set_margin_bottom(16)
        outer.append(btn_row)

        spacer = Gtk.Box()
        spacer.set_hexpand(True)
        btn_row.append(spacer)

        cancel_btn = Gtk.Button(label="Cancel")
        cancel_btn.connect("clicked", lambda _b: self.close())
        btn_row.append(cancel_btn)

        selected_btn = Gtk.Button(label="Reprocess Selected")
        selected_btn.add_css_class("suggested-action")
        selected_btn.connect("clicked", self._on_reprocess_selected)
        btn_row.append(selected_btn)

        all_btn = Gtk.Button(label="Reprocess All")
        all_btn.connect("clicked", self._on_reprocess_all)
        btn_row.append(all_btn)

    def _on_reprocess_selected(self, _btn):
        selected = [path for check, path in self._checks if check.get_active()]
        if not selected:
            return
        self.close()
        self._on_selected(selected, "paste")

    def _on_reprocess_all(self, _btn):
        """Open mode selection dialog before starting batch reprocess."""
        mode_win = _ModeSelectWindow(
            on_mode_selected=self._on_mode_chosen,
            parent=self,
            application=self.get_application(),
        )
        mode_win.present()

    def _on_mode_chosen(self, mode: str):
        self.close()
        self._on_selected(self._failed_wavs, mode)


class _ModeSelectWindow(Gtk.Window):
    """Small modal window asking user to choose batch reprocess mode."""

    def __init__(self, on_mode_selected, parent=None, application=None):
        super().__init__(title="Choose Reprocess Mode", application=application)
        self.set_modal(True)
        self.set_default_size(320, 160)
        if parent:
            self.set_transient_for(parent)

        self._on_mode_selected = on_mode_selected

        box = Gtk.Box(orientation=Gtk.Orientation.VERTICAL, spacing=12)
        box.set_margin_start(24)
        box.set_margin_end(24)
        box.set_margin_top(24)
        box.set_margin_bottom(24)
        self.set_child(box)

        label = Gtk.Label(label="How should reprocessed text be delivered?")
        label.set_wrap(True)
        label.set_xalign(0)
        box.append(label)

        paste_btn = Gtk.Button(label="Paste each into current window")
        paste_btn.add_css_class("suggested-action")
        paste_btn.connect("clicked", lambda _b: self._choose("paste"))
        box.append(paste_btn)

        file_btn = Gtk.Button(label="Write all to file")
        file_btn.connect("clicked", lambda _b: self._choose("file"))
        box.append(file_btn)

    def _choose(self, mode: str):
        self.close()
        self._on_mode_selected(mode)
```

This method is called when the user chooses "Write all to file" in the batch reprocess dialog. It submits all WAV paths to the worker queue with a `batch_file` flag so the worker knows to concatenate results and write to a temp file instead of pasting.

Approach: Add a sentinel in the queue item to differentiate batch-to-file mode from normal paste mode. The simplest implementation is a second queue for batch results + a counter:

Actually, simpler: add a `batch_output_path` field to the queue tuple. When it's set, the worker accumulates transcript to a list and after all batch items complete writes to that path.

For this phase, implement a pragmatic version: `submit_batch_to_file(wav_paths)` submits each WAV individually but each one writes to a shared temp file instead of pasting:

```python
def submit_batch_to_file(self, wav_paths: list[str]) -> str:
    """Submit multiple WAVs for batch reprocess to a text file.

    Returns the output file path. Each WAV is transcribed and post-processed
    in FIFO order; results are written to a temp file which is then opened
    in the default text editor via xdg-open.
    """
    import tempfile
    output_path = tempfile.mktemp(suffix=".txt", prefix="linux-speech-flow-batch-")
    # Clear file first
    open(output_path, "w").close()
    for wav_path in wav_paths:
        config = load_config()
        app_categories = config.get("app_categories", {})
        window_info = get_active_window_info(app_categories=app_categories)
        # Override window_info to use file mode
        window_info["batch_output_path"] = output_path
        self._queue.put((wav_path, window_info, config))
    return output_path
```
