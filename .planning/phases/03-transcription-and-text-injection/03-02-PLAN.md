```python
import os
import subprocess


def get_active_window_info(app_categories: dict | None = None) -> dict:
    """Detect active window and classify app category.

    Returns dict: {
        "window_id": str | None,  # X11 window ID (for --window flag at paste time)
        "title": str,
        "wm_class": str,
        "category": str,          # "terminal", "editor", or "other"
        "session": str,           # "x11" or "wayland"
    }

    On Wayland or detection failure, returns safe defaults with empty strings.
    Capture this at the START of the pipeline (before API calls) to avoid
    focus theft from notifications stealing the window ID.
    """
    session = os.environ.get("XDG_SESSION_TYPE", "x11").lower()
    default = {"window_id": None, "title": "", "wm_class": "", "category": "other", "session": session}

    if session == "wayland":
        return default

    if not os.environ.get("DISPLAY"):
        return default

    try:
        win_id = subprocess.check_output(
            ["xdotool", "getactivewindow"], text=True, timeout=2
        ).strip()
        title = subprocess.check_output(
            ["xdotool", "getwindowname", win_id], text=True, timeout=2
        ).strip()
        xprop_out = subprocess.check_output(
            ["xprop", "-id", win_id, "WM_CLASS"], text=True, timeout=2
        )
        # WM_CLASS(STRING) = "instance", "ClassName"
        parts = xprop_out.split("=")[-1].strip().strip('"').split('", "')
        wm_class = parts[-1].rstrip('"').lower() if parts else ""
        category = _classify(wm_class, app_categories or {})
        return {"window_id": win_id, "title": title, "wm_class": wm_class, "category": category, "session": session}
    except Exception:
        return default


def _classify(wm_class: str, app_categories: dict) -> str:
    """Map wm_class to a category using the user-configurable app_categories dict."""
    terminals = [t.lower() for t in app_categories.get("terminals", [])]
    editors = [e.lower() for e in app_categories.get("editors", [])]
    wm = wm_class.lower()
    if any(t in wm for t in terminals):
        return "terminal"
    if any(e in wm for e in editors):
        return "editor"
    return "other"
```

import os
import subprocess
import time


def paste_text(text: str, window_info: dict) -> None:
    """Write text to clipboard and send paste keystroke to the target window.

    On Wayland: writes to wl-copy clipboard, sends notification to paste manually
    (ydotoold daemon not required). On X11: xclip writes clipboard, xdotool sends
    keystroke to the captured window_id (not current focus — avoids focus-theft bugs).

    The window_id in window_info must be captured BEFORE API calls start.
    """
    session = window_info.get("session", "x11")
    if session == "wayland":
        _wayland_paste(text)
        return

    _x11_paste(text, window_info)


def _x11_paste(text: str, window_info: dict) -> None:
    display = os.environ.get("DISPLAY")
    if not display:
        return

    proc = subprocess.Popen(
        ["xclip", "-selection", "clipboard"],
        stdin=subprocess.PIPE,
        stderr=subprocess.DEVNULL,
    )
    proc.communicate(text.encode("utf-8"))
    time.sleep(0.05)

    window_id = window_info.get("window_id")
    category = window_info.get("category", "other")
    keystroke = "ctrl+shift+v" if category == "terminal" else "ctrl+v"

    cmd = ["xdotool", "key"]
    if window_id:
        cmd += ["--window", window_id]
    cmd.append(keystroke)

    subprocess.run(cmd, check=False, stderr=subprocess.DEVNULL, timeout=3)


def _wayland_paste(text: str) -> None:
    try:
        proc = subprocess.Popen(
            ["wl-copy"],
            stdin=subprocess.PIPE,
            stderr=subprocess.DEVNULL,
        )
        proc.communicate(text.encode("utf-8"))
    except FileNotFoundError:
        pass
    # Keystroke injection requires ydotoold daemon (not required); skip silently.
    # The notify.py send_notification call in transcription.py will inform the user.

Create `src/linux_speech_flow/transcription.py`. This is the central pipeline module. Read the RESEARCH.md patterns carefully before implementing.

Key design decisions (from RESEARCH.md):
- Groq client instantiated with `max_retries=0` — the Fibonacci retry wrapper is the sole retry mechanism. SDK default retries would multiply attempts to 2×5=10 with unpredictable total delay.
- Retry applies PER API call independently: Whisper retries up to 5 times, then LLM retries up to 5 times.
- `get_active_window_info()` is called at `submit()` time (on the GTK main thread), NOT inside the worker thread. This avoids focus-theft from notifications fired during API calls.
- WAV lifecycle: worker thread owns the WAV. On success: os.unlink(). On retry exhaustion: shutil.move() to `~/.local/share/linux-speech-flow/failed/`.
- All GTK side-effects (play_sound, send_notification, on_paste_complete callback) dispatched via `GLib.idle_add()`.
- Silence detection: if `len(transcript.strip()) < 3`, discard WAV, notify "No speech detected", return.
- F9 queue notification: the `submit()` method returns the queue depth; the caller (app.py) is responsible for showing "Recording queued (N pending)" if depth > 1.

```python
import os
import queue
import shutil
import threading
import time
from datetime import datetime
from pathlib import Path
from uuid import uuid4

import groq
from gi.repository import GLib

from linux_speech_flow.config import load_config
from linux_speech_flow.injector import paste_text
from linux_speech_flow.notify import send_notification
from linux_speech_flow.sounds import play_sound
from linux_speech_flow.window_context import get_active_window_info

FIBONACCI_DELAYS = [5, 8, 13, 21, 34]
FAILED_DIR = Path.home() / ".local" / "share" / "linux-speech-flow" / "failed"
MIN_TRANSCRIPT_LEN = 3


def _failed_wav_name() -> str:
    return f"recording_{datetime.now().strftime('%Y%m%dT%H%M%S')}_{uuid4().hex[:4]}.wav"


def _call_with_retry(fn, *args, retryable, **kwargs):
    """Call fn(*args, **kwargs), retrying on retryable exceptions with Fibonacci backoff.
    Raises the last exception after all delays are exhausted.
    """
    last_exc = None
    for delay in FIBONACCI_DELAYS:
        try:
            return fn(*args, **kwargs)
        except retryable as exc:
            last_exc = exc
            time.sleep(delay)
    raise last_exc


def _classify_groq_error(exc: groq.APIError) -> str:
    if isinstance(exc, groq.AuthenticationError):
        return "Invalid API key — check Settings"
    if isinstance(exc, groq.RateLimitError):
        return "Rate limit exceeded — try again shortly"
    if isinstance(exc, groq.APIConnectionError):
        return "Network error — check your internet connection"
    return f"Groq API error: {getattr(exc, 'status_code', 'unknown')}"


def _build_user_message(raw_transcript: str, window_info: dict, vocabulary: list) -> str:
    parts = []
    if window_info.get("wm_class") or window_info.get("title"):
        parts.append(
            f"<context>\n"
            f"Application: {window_info.get('wm_class', '')}\n"
            f"Window title: {window_info.get('title', '')}\n"
            f"App category: {window_info.get('category', 'other')}\n"
            f"</context>\n"
        )
    if vocabulary:
        vocab_str = "\n".join(f"- {word}" for word in vocabulary)
        parts.append(
            f"<vocabulary>\n"
            f"The following words or phrases are correct and must not be altered:\n"
            f"{vocab_str}\n"
            f"</vocabulary>\n"
        )
    parts.append(f"Transcript:\n{raw_transcript}")
    return "\n".join(parts)


class TranscriptionPipeline:
    """Background pipeline: WAV → Whisper → LLM → clipboard paste.

    Threading contract:
    - submit() is called from the GTK main thread; it captures window context
      immediately (before API calls) to prevent focus-theft by notifications.
    - _run() runs on a daemon worker thread; all GTK side-effects go via GLib.idle_add().
    - One worker thread processes the queue serially (FIFO). F9 presses during
      active pipeline add to the queue; the caller shows a "Recording queued" notification.
    """

    def __init__(self, on_paste_complete=None, on_error=None):
        self._queue: queue.Queue = queue.Queue()
        self._on_paste_complete = on_paste_complete
        self._on_error = on_error
        self._worker = threading.Thread(target=self._run, daemon=True, name="transcription-worker")
        self._worker.start()

    def submit(self, wav_path: str) -> int:
        """Queue wav_path for processing. Returns queue depth AFTER insert.

        Captures window context NOW (on GTK main thread) to avoid focus-theft
        by notifications fired during API calls. Returns depth so caller can
        notify user if depth > 1 ("Recording queued (N pending)").
        """
        config = load_config()
        app_categories = config.get("app_categories", {})
        window_info = get_active_window_info(app_categories=app_categories)
        self._queue.put((wav_path, window_info, config))
        return self._queue.qsize()

    def _run(self):
        while True:
            item = self._queue.get()
            try:
                self._process(*item)
            except Exception as exc:
                GLib.idle_add(self._dispatch_error, str(exc))
            finally:
                self._queue.task_done()

    def _process(self, wav_path: str, window_info: dict, config: dict):
        api_key = config.get("groq_api_key", "")
        client = groq.Groq(api_key=api_key, max_retries=0)

        sounds_enabled = config.get("sounds_enabled", True)
        output_device = config.get("sounds_output_device", "")
        processing_enabled = config.get("processing_sound_enabled", True)
        success_enabled = config.get("success_sound_enabled", True)
        timeout = config.get("pipeline_timeout", 60)
        whisper_model = config.get("whisper_model", "whisper-large-v3-turbo")
        llm_model = config.get("llm_model", "meta-llama/llama-4-scout-17b-16e-instruct")
        system_prompt = config.get("llm_system_prompt", "")
        vocabulary = config.get("vocabulary", [])

        # Play processing sound (on GTK thread via idle_add)
        if sounds_enabled and processing_enabled:
            GLib.idle_add(play_sound, "processing.wav", output_device, True)

        # Step 1: Whisper transcription with Fibonacci retry
        retryable_errors = (groq.APIConnectionError, groq.RateLimitError)
        try:
            raw_transcript = _call_with_retry(
                self._transcribe, client, wav_path, whisper_model,
                retryable=retryable_errors,
            )
        except groq.AuthenticationError as exc:
            msg = _classify_groq_error(exc)
            self._save_failed_wav(wav_path)
            GLib.idle_add(self._dispatch_api_error, msg, wav_path)
            return
        except Exception as exc:
            msg = _classify_groq_error(exc) if isinstance(exc, groq.APIError) else str(exc)
            self._save_failed_wav(wav_path)
            GLib.idle_add(self._dispatch_api_error, msg, wav_path)
            return

        # Check for silence / too-short transcript
        if len(raw_transcript.strip()) < MIN_TRANSCRIPT_LEN:
            try:
                os.unlink(wav_path)
            except OSError:
                pass
            GLib.idle_add(send_notification, "No speech detected", "Recording was too short or silent.")
            return

        # Step 2: LLM post-processing with Fibonacci retry
        final_text = raw_transcript
        llm_failed = False
        try:
            user_message = _build_user_message(raw_transcript, window_info, vocabulary)
            final_text = _call_with_retry(
                self._postprocess, client, user_message, system_prompt, llm_model,
                retryable=retryable_errors,
            )
        except Exception:
            llm_failed = True
            final_text = raw_transcript  # TRANS-05: fallback to raw transcript

        # Step 3: Paste text
        paste_text(final_text, window_info)

        # Step 4: WAV cleanup on success
        try:
            os.unlink(wav_path)
        except OSError:
            pass

        # Step 5: Notify and play success sound
        if llm_failed:
            GLib.idle_add(send_notification, "LLM failed — raw transcript pasted", "")
        elif window_info.get("session") == "wayland":
            GLib.idle_add(send_notification, "Text copied to clipboard", "Press Ctrl+V to paste.")

        if sounds_enabled and success_enabled and not llm_failed:
            GLib.idle_add(play_sound, "success.wav", output_device, True)
        elif sounds_enabled and success_enabled and llm_failed:
            GLib.idle_add(play_sound, "success.wav", output_device, True)

        if self._on_paste_complete:
            GLib.idle_add(self._on_paste_complete)

    def _transcribe(self, client: groq.Groq, wav_path: str, model: str) -> str:
        with open(wav_path, "rb") as f:
            result = client.audio.transcriptions.create(
                file=f,
                model=model,
                response_format="text",
                language="en",
                temperature=0.0,
            )
        return result.strip() if isinstance(result, str) else str(result).strip()

    def _postprocess(self, client: groq.Groq, user_message: str, system_prompt: str, model: str) -> str:
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_message},
            ],
            temperature=0.0,
            max_tokens=2048,
        )
        return response.choices[0].message.content.strip()

    def _save_failed_wav(self, wav_path: str) -> str | None:
        """Move WAV to failed/ directory. Returns new path or None on error."""
        try:
            FAILED_DIR.mkdir(parents=True, exist_ok=True)
            dest = FAILED_DIR / _failed_wav_name()
            shutil.move(wav_path, dest)
            return str(dest)
        except Exception:
            return None

    def _dispatch_api_error(self, message: str, wav_path: str):
        failed_note = f"\nFailed WAV saved to {FAILED_DIR}. Press F10 to reprocess."
        send_notification("Transcription failed", message + failed_note)
        if self._on_error:
            self._on_error(message)
        return False

    def _dispatch_error(self, message: str):
        send_notification("Pipeline error", message)
        if self._on_error:
            self._on_error(message)
        return False
```
