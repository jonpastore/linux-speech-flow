    Add `_handle_capture_key`:
    ```python
    def _handle_capture_key(self, keyval: int, state) -> bool:
        from gi.repository import Gdk as _Gdk
        global _GTK_MODIFIER_KEYSYMS
        try:
            _GTK_MODIFIER_KEYSYMS
        except NameError:
            _GTK_MODIFIER_KEYSYMS = frozenset({
                _Gdk.KEY_Control_L, _Gdk.KEY_Control_R,
                _Gdk.KEY_Alt_L, _Gdk.KEY_Alt_R,
                _Gdk.KEY_Shift_L, _Gdk.KEY_Shift_R,
                _Gdk.KEY_Super_L, _Gdk.KEY_Super_R,
                _Gdk.KEY_ISO_Level3_Shift,
                _Gdk.KEY_Caps_Lock, _Gdk.KEY_Num_Lock,
            })

        if keyval == _Gdk.KEY_Escape:
            self._cancel_capture()
            return True

        # Ignore bare modifier presses
        if keyval in _GTK_MODIFIER_KEYSYMS:
            return True

        # Build modifier set from GDK state
        mods = set()
        if state & _Gdk.ModifierType.CONTROL_MASK: mods.add('ctrl')
        if state & _Gdk.ModifierType.ALT_MASK:     mods.add('alt')
        if state & _Gdk.ModifierType.SHIFT_MASK:   mods.add('shift')
        if state & _Gdk.ModifierType.SUPER_MASK:   mods.add('super')

        # Require at least one modifier
        if not mods:
            self._hotkey_error_label.set_text("Combo must include at least one modifier (Ctrl, Alt, Shift, Super)")
            return True

        # Build key identifier
        key_id = self._gdk_keyval_to_id(keyval)
        if key_id is None:
            return True

        # Serialize in canonical order
        mod_order = ['ctrl', 'alt', 'shift', 'super']
        combo_str = '+'.join(m for m in mod_order if m in mods) + '+' + key_id
        self._accept_capture(combo_str)
        return True
    ```

    Add `_gdk_keyval_to_id` as a static method:

    ```python
    @staticmethod
    def _gdk_keyval_to_id(keyval: int) -> str | None:
        from gi.repository import Gdk as _Gdk
        _SPECIAL = {
            _Gdk.KEY_Escape: 'esc',
            _Gdk.KEY_Delete: 'delete',
            _Gdk.KEY_Return: 'enter',
            _Gdk.KEY_Tab: 'tab',
            _Gdk.KEY_space: 'space',
            _Gdk.KEY_Left: 'left',   _Gdk.KEY_Right: 'right',
            _Gdk.KEY_Up: 'up',       _Gdk.KEY_Down: 'down',
            _Gdk.KEY_Home: 'home',   _Gdk.KEY_End: 'end',
            _Gdk.KEY_Page_Up: 'page_up', _Gdk.KEY_Page_Down: 'page_down',
            _Gdk.KEY_Insert: 'insert',
            _Gdk.KEY_F1: 'f1',   _Gdk.KEY_F2: 'f2',   _Gdk.KEY_F3: 'f3',
            _Gdk.KEY_F4: 'f4',   _Gdk.KEY_F5: 'f5',   _Gdk.KEY_F6: 'f6',
            _Gdk.KEY_F7: 'f7',   _Gdk.KEY_F8: 'f8',   _Gdk.KEY_F9: 'f9',
            _Gdk.KEY_F10: 'f10', _Gdk.KEY_F11: 'f11', _Gdk.KEY_F12: 'f12',
        }
        if keyval in _SPECIAL:
            return _SPECIAL[keyval]
        char = chr(keyval).lower()
        if char.isalpha() or char.isdigit():
            return char
        return None
    ```

    Add `_accept_capture`:
    ```python
    def _accept_capture(self, combo_str: str) -> None:
        action = self._capture_action
        if action is None:
            return

        # Check dangerous combos
        if combo_str in DANGEROUS_COMBOS:
            from linux_speech_flow.hotkey import combo_display as _cd
            self._hotkey_error_label.set_text(
                f"{_cd(combo_str)} is reserved by the system"
            )
            self._cancel_capture()
            return

        # Check conflicts with other actions
        for other_action, current_combo in self._hotkey_values.items():
            if other_action == action:
                continue
            if current_combo == combo_str:
                from linux_speech_flow.hotkey import combo_display as _cd
                self._hotkey_error_label.set_text(
                    f"{_cd(combo_str)} is already used for {HOTKEY_ACTION_LABELS[other_action]}"
                )
                self._cancel_capture()
                return

        self._hotkey_error_label.set_text("")
        self._capture_action = None
        self._apply_binding(action, combo_str)
    ```

    IMPLEMENTATION NOTES:
    - Insertion point for Hotkeys section: after the Gemini section (after `self._gemini_status` label) and BEFORE the existing `sep1` separator that starts the Vocabulary section. This places Hotkeys right after AI Integrations as specified in CONTEXT.md.
    - The `_GTK_MODIFIER_KEYSYMS` global initialization is lazy (inside _handle_capture_key) to avoid importing Gdk at module level before gi.require_version is called. Use a try/except NameError pattern.
    - `reload_bindings()` in `_apply_binding` writes the new combo into `_hotkey_values` BEFORE calling reload — so reload reads from config.json, not from `_hotkey_values`. This means `_apply_binding` must NOT pre-emptively write to config; the write happens only in `_on_save`. Instead, `reload_bindings()` re-reads config and the NEW value isn't persisted yet. This creates a UX quirk: the button shows the new binding, but if the app restarts before Save, it reverts. This is acceptable per CONTEXT.md ("Save button persists bindings"). However, `reload_bindings()` must use the in-memory state, not config, to enable the "changes take effect immediately" requirement. Solution: pass the current in-memory bindings dict to the hotkey manager OR reload from a temp config dict. PREFERRED APPROACH: `_apply_binding` calls `self._hotkey_manager.apply_binding_override(action, combo_str)` which updates the in-memory `_bindings` dict directly without touching config. Add `apply_binding_override` to HotkeyManager in Task 2 (or as an addendum here):

    ```python
    def apply_binding_override(self, action: str, combo_str: str) -> None:
        """Apply a single binding immediately without requiring a config save."""
    ```
