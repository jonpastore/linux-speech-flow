    ```python
    import logging
    import threading
    from gi.repository import GLib
    from slack_sdk.socket_mode import SocketModeClient
    from slack_sdk.socket_mode.response import SocketModeResponse
    from slack_sdk.web import WebClient

    logger = logging.getLogger(__name__)


    class SlackSocket:
        """SocketModeClient wrapper for Slack huddle event detection.

        Runs SocketModeClient.connect() in a daemon thread.
        Never calls start() — it blocks the calling thread.
        All GTK callbacks dispatched via GLib.idle_add.
        """

        def __init__(self):
            self._client: SocketModeClient | None = None

        def start(
            self,
            app_token: str,
            bot_token: str,
            on_huddle_event,
            authed_user_id: str,
        ) -> None:
            """Connect to Slack Socket Mode in a daemon thread."""
            self._client = SocketModeClient(
                app_token=app_token,
                web_client=WebClient(token=bot_token),
            )
            self._client.socket_mode_request_listeners.append(
                self._make_listener(on_huddle_event, authed_user_id)
            )
            threading.Thread(
                target=self._client.connect,
                daemon=True,
                name="slack-socket-mode",
            ).start()
            logger.info("SlackSocket: connecting via Socket Mode (daemon thread)")

        def stop(self) -> None:
            if self._client:
                try:
                    self._client.close()
                except Exception:
                    pass
                self._client = None

        def _make_listener(self, on_huddle_event, authed_user_id: str):
            def listener(client: SocketModeClient, req):
                try:
                    if req.type == "events_api":
                        payload = req.payload or {}
                        event = payload.get("event", {})
                        if event.get("type") == "user_huddle_changed":
                            event_user = event.get("user", {})
                            user_id = event_user.get("id") if isinstance(event_user, dict) else event_user
                            if user_id == authed_user_id:
                                GLib.idle_add(on_huddle_event, event)
                            else:
                                logger.debug("Ignoring huddle event for user %s (not authed user)", user_id)
                finally:
                    client.send_socket_mode_response(
                        SocketModeResponse(envelope_id=req.envelope_id)
                    )
            return listener
    ```

    ```python
    import logging
    import re
    import threading

    from gi.repository import GLib

    logger = logging.getLogger(__name__)

    # All 11 supported commands (sorted by length descending for prefix-match correctness)
    _COMMANDS = [
        "start recording",
        "stop recording",
        "list action items",
        "pause",
        "resume",
        "summarize",
        "calibrate",
        "status",
        "help",
        # "note" and "topic" handled via prefix match below
    ]

    WELCOME_MESSAGE = (
        "Hi, I'm recording this huddle.\n"
        "Available commands:\n"
        "  conyo start/stop recording | conyo pause/resume | conyo summarize\n"
        "  conyo calibrate | conyo status | conyo list action items\n"
        "  conyo note [text] | conyo topic [title] | conyo help\n"
        "(Replace 'conyo' with your configured activation word.)"
    )


    def detect_activation(text: str, word: str) -> tuple[str | None, str | None]:
        """Scan transcript text for activation word followed by a command.

        Returns (command_key, remainder_text) or (None, None) if no match.
        command_key is one of the _COMMANDS list, 'note', or 'topic'.
        """
        pattern = re.compile(
            rf'\b{re.escape(word.lower())}\s+(.+)',
            re.IGNORECASE,
        )
        m = pattern.search(text.lower())
        if not m:
            return None, None
        remainder = m.group(1).strip()
        for cmd in sorted(_COMMANDS, key=len, reverse=True):
            if remainder.startswith(cmd):
                after = remainder[len(cmd):].strip()
                return cmd, after
        if remainder.startswith("note "):
            return "note", remainder[5:].strip()
        if remainder.startswith("topic "):
            return "topic", remainder[6:].strip()
        return None, None
    ```

    HuddleManager owns: HuddleRecorder + HuddleStatusWindow + per-chunk transcription threads + activation word dispatch + Slack bot messaging + confidence alerting.

    ```python
    class HuddleManager:
        """Orchestrates Slack huddle recording sessions.

        Threading contract:
        - start_session() / stop_session() called on GTK main thread.
        - Per-chunk Whisper transcription runs in daemon threads.
        - Chunk callbacks dispatched via GLib.idle_add (GTK main thread).
        - All GTK / HuddleStatusWindow operations from GTK main thread only.
        - on_session_complete fires on GTK main thread.
        """

        def __init__(self, application, slack_manager, on_session_complete, on_tray_state):
            """
            Args:
                application: Gtk.Application (for HuddleStatusWindow creation)
                slack_manager: SlackManager instance
                on_session_complete: Callable(transcript, metadata) — fires on session end
                on_tray_state: Callable(state: str) — 'huddle' or 'idle'
            """
            self._app = application
            self._slack_manager = slack_manager
            self._on_session_complete = on_session_complete
            self._on_tray_state = on_tray_state
            self._recorder: HuddleRecorder | None = None
            self._status_window = None
            self._chunk_texts: list[str] = []
            self._chunk_count = 0
            self._in_flight = 0
            self._session_ending = False
            self._active_team_id: str | None = None
            self._active_channel_id: str | None = None
    ```

    Key methods to implement:

    **`start_session(team_id, channel_id)`** — GTK main thread:
    1. Load config: mic device, chunk_silence_sec, silence_rms_threshold, slack_activation_word, slack_confidence_threshold
    2. Create `HuddleRecorder(mic_device=config['microphone'])` and call `.start(on_chunk_ready=self._on_chunk_ready, on_error=self._on_recorder_error)`
    3. Create `HuddleStatusWindow(application=self._app)`, call `.present()`, call `.start_elapsed_timer()`
    4. Update Slack status dot in window: `_status_window.update_slack_status(True, team_name, channel_id)`
    5. Call `on_tray_state('huddle')`
    6. Post welcome message to Slack on a background thread: `threading.Thread(target=self._post_welcome, args=(team_id, channel_id), daemon=True).start()`

    **`stop_session()`** — GTK main thread:
    1. Set `_session_ending = True`
    2. Stop HuddleRecorder
    3. Stop HuddleStatusWindow elapsed timer, close window
    4. Call `on_tray_state('idle')`
    5. If no in-flight threads, call `_finish_session()` immediately. Else drain (same pattern as ConversationManager: `GLib.timeout_add(500, self._drain_check)`)

    **`_on_chunk_ready(wav_path)`** — GTK main thread (via GLib.idle_add from HuddleRecorder):
    1. `_in_flight += 1`, `_chunk_count += 1`
    2. Update chunk count in status window
    3. Dispatch transcription in daemon thread: `threading.Thread(target=self._transcribe_chunk, args=(wav_path,), daemon=True).start()`

    **`_transcribe_chunk(wav_path)`** — worker thread:
    1. Call Groq Whisper with `response_format="verbose_json"` to get text + confidence
    2. Compute confidence from avg_logprob (formula from RESEARCH.md)
    3. Run detect_activation(text, config['slack_activation_word'])
    4. If command found: GLib.idle_add(self._dispatch_command, cmd_key, remainder, team_id, channel_id) — do NOT add chunk to transcript
    5. If no command: append text to _chunk_texts, GLib.idle_add(self._on_chunk_transcribed, text, confidence)
    6. GLib.idle_add(self._on_in_flight_done)

    **`_on_chunk_transcribed(text, confidence)`** — GTK main thread (via GLib.idle_add):
    1. Update status window: `update_transcript(text)`, `update_confidence(confidence)`
    2. If confidence < threshold: dispatch confidence alert on background thread

    **`_dispatch_command(cmd_key, remainder, team_id, channel_id)`** — GTK main thread:
    1. Update status window: `update_last_command(f"conyo {cmd_key}")`, `set_command_processing(True)`
    2. Run command in daemon thread; on completion, GLib.idle_add to `set_command_processing(False)`
    3. Command implementations:
       - `stop recording`: calls `stop_session()`
       - `pause`: sets internal paused flag, stops HuddleRecorder but keeps window open
       - `resume`: restarts HuddleRecorder
       - `summarize`: calls Groq LLM on current transcript text, posts result to Slack
       - `calibrate`: posts "Please speak one at a time and closer to your mic" to Slack channel
       - `status`: posts current recording duration + chunk count to Slack
       - `list action items`: calls Groq LLM, posts action items to Slack
       - `note [text]`: appends "[NOTE: {remainder}]" to _chunk_texts, posts to Slack
       - `topic [title]`: appends "\n## {remainder}\n" to _chunk_texts, posts to Slack
       - `help`: calls `_slack_manager.post_message()` with WELCOME_MESSAGE
       - `start recording`: ignored if already recording (log warning)

    **`_post_confidence_alert(team_id, channel_id)`** — worker thread:
    1. `_slack_manager.post_message(team_id, channel_id, "Having trouble understanding — please try speaking one at a time")`
    2. `send_notification("Low transcription confidence in huddle", body="Check Slack for alert")`

    **`_finish_session()`** — GTK main thread:
    Full transcript = '\n'.join(_chunk_texts); metadata = {date, duration, chunk_count, session_type: 'huddle', team_id, channel_id}. Call `on_session_complete(transcript, metadata)`.

    **`is_active(self) -> bool`** — public property:
    Return `self._recorder is not None`. Used by tray toggle callback to check if a huddle is in progress without accessing private state.

    Import pattern for circular import avoidance:
    - Import `HuddleRecorder` at module top (not circular)
    - Import `HuddleStatusWindow` lazily inside `start_session()`: `from linux_speech_flow.huddle_status import HuddleStatusWindow`
    - Import `groq` inside `_transcribe_chunk` if needed, or import `load_config` and construct client there
