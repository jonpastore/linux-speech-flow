**Step 3: Create src/linux_speech_flow/scripts/generate_sounds.py** — a dev-only script. When executed it writes the 3 WAV files into `src/linux_speech_flow/sounds/`. Uses only stdlib (wave, math, struct). Sample rate: 22050 Hz, mono, s16le, amplitude 0.3.

Sound design (per CONTEXT.md):
- `start.wav`: ascending two-tone chime — 440 Hz for 0.15s followed by 880 Hz for 0.15s, 10ms fade in/out on each tone, total 0.3s
- `stop.wav`: descending two-tone chime — 880 Hz for 0.15s followed by 440 Hz for 0.15s, 10ms fade in/out on each tone, total 0.3s
- `error.wav`: triple short buzz — 200 Hz × 3 pulses (each 0.1s on + 0.05s gap), total ~0.45s, 5ms fade in/out on each buzz

Implementation:
```python
#!/usr/bin/env python3
"""Dev script: generate bundled WAV sound files for linux-speech-flow.
Run from the repo root: python src/linux_speech_flow/scripts/generate_sounds.py
"""
import math
import os
import struct
import wave
from pathlib import Path

SAMPLE_RATE = 22050
AMPLITUDE = 0.3
OUT_DIR = Path(__file__).parent.parent / "sounds"


def generate_tone(freq_hz: float, duration_s: float, fade_s: float = 0.01) -> bytes:
    n = int(SAMPLE_RATE * duration_s)
    fade = int(SAMPLE_RATE * fade_s)
    samples = []
    for i in range(n):
        t = i / SAMPLE_RATE
        val = AMPLITUDE * math.sin(2 * math.pi * freq_hz * t)
        if i < fade:
            val *= i / fade
        elif i > n - fade:
            val *= (n - i) / fade
        samples.append(max(-32767, min(32767, int(val * 32767))))
    return struct.pack(f"{n}h", *samples)


def write_wav(path: Path, data: bytes) -> None:
    with wave.open(str(path), "wb") as wf:
        wf.setnchannels(1)
        wf.setsampwidth(2)
        wf.setframerate(SAMPLE_RATE)
        wf.writeframes(data)


def main():
    OUT_DIR.mkdir(parents=True, exist_ok=True)

    # start.wav: ascending chime (440 Hz -> 880 Hz)
    start_data = generate_tone(440, 0.15) + generate_tone(880, 0.15)
    write_wav(OUT_DIR / "start.wav", start_data)
    print(f"Written: {OUT_DIR / 'start.wav'}")

    # stop.wav: descending chime (880 Hz -> 440 Hz)
    stop_data = generate_tone(880, 0.15) + generate_tone(440, 0.15)
    write_wav(OUT_DIR / "stop.wav", stop_data)
    print(f"Written: {OUT_DIR / 'stop.wav'}")

    # error.wav: triple buzz (200 Hz x3 with gaps)
    buzz = generate_tone(200, 0.1, fade_s=0.005)
    silence = bytes(int(SAMPLE_RATE * 0.05) * 2)  # 50ms silence
    error_data = buzz + silence + buzz + silence + buzz
    write_wav(OUT_DIR / "error.wav", error_data)
    print(f"Written: {OUT_DIR / 'error.wav'}")

    print("Done.")


if __name__ == "__main__":
    main()
```

**sounds.py** — non-blocking paplay wrapper using importlib.resources for bundled file paths:

```python
import importlib.resources
import subprocess


def play_sound(sound_name: str, output_device: str | None = None, enabled: bool = True) -> None:
    """Play a bundled sound file non-blocking via paplay.

    Args:
        sound_name: Filename in linux_speech_flow/sounds/ (e.g. 'start.wav').
        output_device: PulseAudio sink name. None or empty string = system default.
        enabled: If False, this function is a no-op (sounds_enabled=False in config).
    """
    if not enabled:
        return
    ref = importlib.resources.files("linux_speech_flow.sounds").joinpath(sound_name)
    with importlib.resources.as_file(ref) as path:
        cmd = ["paplay"]
        if output_device:
            cmd.append(f"--device={output_device}")
        cmd.append(str(path))
        subprocess.Popen(cmd, stderr=subprocess.DEVNULL)
```

Important notes:
- Use `importlib.resources.files()` (NOT `__file__`-relative paths) so it works correctly in .deb/wheel installations where source files may be in a zip.
- `subprocess.Popen` (NOT `subprocess.run`) so playback is non-blocking — returns immediately while paplay plays in background.
- `stderr=subprocess.DEVNULL` suppresses paplay warnings if device is temporarily unavailable.
- If `paplay` binary is missing, `Popen` raises `FileNotFoundError` — do NOT catch this; it propagates as a programming environment issue, not a user-facing error.

**notify.py** — notify-send wrapper that captures and returns the notification ID for Phase 3's replace-chain:

```python
import subprocess


def send_notification(
    summary: str,
    body: str = "",
    replace_id: int | None = None,
) -> int | None:
    """Send a desktop notification via notify-send.

    Uses -p (print-id) to capture the notification ID, enabling Phase 3
    to replace "Recording..." with "Transcribing..." via notify-send -r <id>.

    Args:
        summary: Notification title.
        body: Optional notification body text.
        replace_id: If provided, replaces an existing notification with this ID.

    Returns:
        Notification ID as int, or None if notify-send is unavailable or -p
        is not supported (libnotify < 0.7.9).
    """
    cmd = ["notify-send", "-p"]  # -p prints the ID to stdout
    if replace_id is not None:
        cmd += ["-r", str(replace_id)]
    cmd.append(summary)
    if body:
        cmd.append(body)
    try:
        result = subprocess.run(
            cmd, capture_output=True, text=True, timeout=2
        )
        stripped = result.stdout.strip()
        return int(stripped) if stripped else None
    except (subprocess.TimeoutExpired, ValueError, FileNotFoundError, OSError):
        return None
```

Important notes:
- `-p` flag (print-id) requires libnotify >= 0.7.9, which Ubuntu 22.04/Pop!_OS 22.04 ship.
- Returns None gracefully if notify-send is missing or too old — app continues working, just no notification ID for replace-chain.
- `timeout=2` prevents blocking indefinitely if D-Bus is unresponsive.
- Do NOT use `Popen` here — we need to capture stdout synchronously to get the ID.
