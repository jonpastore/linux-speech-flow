  <action>
Create `src/linux_speech_flow/hotkey.py` with `HotkeyManager`.

**State machine:**
- IDLE: waiting for F9 press
- WAITING: F9 is down, measuring hold duration (debounce 300ms)
- RECORDING: audio capture active

**State transitions:**
- IDLE + F9_DOWN → record press_time via time.monotonic(), enter WAITING
- WAITING + F9_UP (< 300ms) → discard (brief tap), return to IDLE
- WAITING + F9_UP (>= 300ms) → call _start_recording(), enter RECORDING
- WAITING + 300ms elapsed (GLib.timeout_add fires) → call _start_recording(), enter RECORDING
- RECORDING + F9_UP → call _stop_recording(cancel=False)
- RECORDING + ESC_DOWN → call _stop_recording(cancel=True)
- (AudioRecorder fires on_complete → _on_recorder_complete → plays stop.wav → IDLE → calls _on_complete_cb)
- (AudioRecorder fires on_error → _on_recorder_error → plays error.wav → IDLE)

**Implementation:**

```python
import time
import threading
from pynput import keyboard
from gi.repository import GLib

from linux_speech_flow.config import load_config
from linux_speech_flow.recorder import AudioRecorder
from linux_speech_flow.sounds import play_sound
from linux_speech_flow.notify import send_notification


class HotkeyManager:
    """Global F9 hold-to-record hotkey with push-to-talk state machine.

    Threading contract:
    - pynput Listener callbacks (_on_press, _on_release) fire from the Listener
      daemon thread. They must NOT call GTK/GLib functions directly.
    - All GTK operations (_start_recording, _stop_recording, etc.) are
      dispatched via GLib.idle_add() to run on the GTK main thread.
    - AudioRecorder callbacks (on_complete, on_error) already arrive on the
      GTK main thread via GLib.idle_add in recorder.py.
    """

    _STATE_IDLE = "idle"
    _STATE_WAITING = "waiting"
    _STATE_RECORDING = "recording"
    _DEBOUNCE_MS = 300  # minimum hold duration in milliseconds

    def __init__(self, on_recording_complete, on_recording_error=None):
        """
        Args:
            on_recording_complete: Callable(wav_path: str) — called on GTK main
                thread when recording finishes normally. wav_path is ready for
                Phase 3 transcription. May be None for Phase 2 standalone testing.
            on_recording_error: Optional callable(message: str) — called on GTK
                main thread on mic error. If None, errors are shown but not propagated.
        """
        self._on_complete_cb = on_recording_complete
        self._on_error_cb = on_recording_error
        self._state = self._STATE_IDLE
        self._press_time: float | None = None
        self._recorder: AudioRecorder | None = None
        self._notif_id: int | None = None
        self._started = False  # guard: ignore F9 held at startup

        self._listener = keyboard.Listener(
            on_press=self._on_press,
            on_release=self._on_release,
        )
        self._listener.daemon = True
        # Do NOT use suppress=True — crashes X11 sessions (pynput issue #269)

    def start(self) -> None:
        """Start the global keyboard listener. Call from do_startup()."""
        self._listener.start()

    def stop(self) -> None:
        """Stop the listener. Safe to call at app shutdown."""
        self._listener.stop()

    def mark_started(self) -> bool:
        """Called via GLib.idle_add after first GTK main loop tick.
        Guards against F9 being held when the app starts.
        Returns False so GLib.idle_add removes this callback.
        """
        self._started = True
        return False

    # --- pynput callbacks (Listener thread) ---

    def _on_press(self, key):
        if not self._started:
            return
        if key == keyboard.Key.f9 and self._state == self._STATE_IDLE:
            self._press_time = time.monotonic()
            self._state = self._STATE_WAITING
            # Schedule debounce check: if still WAITING after 300ms, start recording
            GLib.idle_add(self._schedule_debounce_timer)
        elif key == keyboard.Key.esc and self._state == self._STATE_RECORDING:
            GLib.idle_add(self._stop_recording, True)  # cancel=True

    def _on_release(self, key):
        if not self._started:
            return
        if key == keyboard.Key.f9:
            if self._state == self._STATE_WAITING:
                elapsed_ms = (time.monotonic() - self._press_time) * 1000
                if elapsed_ms >= self._DEBOUNCE_MS:
                    GLib.idle_add(self._start_recording)
                else:
                    self._state = self._STATE_IDLE  # brief tap — discard
            elif self._state == self._STATE_RECORDING:
                GLib.idle_add(self._stop_recording, False)  # cancel=False

    # --- GTK main thread methods (called via GLib.idle_add) ---

    def _schedule_debounce_timer(self) -> bool:
        """Schedule a 300ms timer to start recording if F9 still held.
        Called via GLib.idle_add from Listener thread.
        Returns False (remove from idle queue).
        """
        GLib.timeout_add(self._DEBOUNCE_MS, self._on_debounce_elapsed)
        return False

    def _on_debounce_elapsed(self) -> bool:
        """Fires 300ms after F9 press. Start recording if still WAITING."""
        if self._state == self._STATE_WAITING:
            self._start_recording()
        return False  # do not repeat timer

    def _start_recording(self) -> bool:
        """Start audio capture. Called on GTK main thread."""
        if self._state != self._STATE_WAITING:
            return False  # state changed between schedule and execution
        self._state = self._STATE_RECORDING

        config = load_config()
        sounds_enabled = config.get("sounds_enabled", True)
        output_device = config.get("sounds_output_device", "")
        device_name = config.get("microphone", "")
        max_dur = config.get("max_recording_duration", 300)
        silence_dur = config.get("silence_stop_duration", 10)

        play_sound("start.wav", output_device=output_device, enabled=sounds_enabled)
        self._notif_id = send_notification("Recording...")

        self._recorder = AudioRecorder(
            device_name=device_name,
            max_duration=max_dur,
            silence_duration=silence_dur,
        )
        self._recorder.start(
            on_complete=self._on_recorder_complete,
            on_error=self._on_recorder_error,
        )
        return False

    def _stop_recording(self, cancel: bool) -> bool:
        """Stop or cancel an in-progress recording. Called on GTK main thread."""
        if self._state != self._STATE_RECORDING:
            return False
        if self._recorder:
            self._recorder.stop(cancel=cancel)
            self._recorder = None

        config = load_config()
        sounds_enabled = config.get("sounds_enabled", True)
        output_device = config.get("sounds_output_device", "")
        play_sound("stop.wav", output_device=output_device, enabled=sounds_enabled)

        # Reset state now; on_complete/on_error callbacks handle the rest
        self._state = self._STATE_IDLE
        return False

    def _on_recorder_complete(self, wav_path: str) -> bool:
        """Called on GTK main thread by AudioRecorder when WAV is ready.

        This fires for ALL normal completion paths: silence auto-stop and
        max-duration auto-stop. Per CONTEXT.md Area 2, normal stop always
        plays the descending stop chime. Load config here the same way
        _stop_recording does so sounds_enabled and sounds_output_device are
        respected on auto-stop paths too.
        """
        config = load_config()
        sounds_enabled = config.get("sounds_enabled", True)
        output_device = config.get("sounds_output_device", "")
        play_sound("stop.wav", output_device=output_device, enabled=sounds_enabled)

        self._state = self._STATE_IDLE
        if self._on_complete_cb:
            self._on_complete_cb(wav_path)
        return False

    def _on_recorder_error(self, message: str) -> bool:
        """Called on GTK main thread by AudioRecorder on mic error."""
        self._state = self._STATE_IDLE
        self._recorder = None

        config = load_config()
        sounds_enabled = config.get("sounds_enabled", True)
        output_device = config.get("sounds_output_device", "")
        play_sound("error.wav", output_device=output_device, enabled=sounds_enabled)
        send_notification(
            "Microphone unavailable — check Settings",
            body=message,
        )
        if self._on_error_cb:
            self._on_error_cb(message)
        return False
```

Important notes:
- `_stop_recording` handles the F9-release and ESC cancel paths — it plays stop.wav and resets state to IDLE immediately. The recorder's stop() call is non-blocking (just sets threading.Event). The WAV file on cancel is deleted by the recorder thread.
- `_on_recorder_complete` handles the auto-stop paths (silence timeout, max duration). It MUST play stop.wav before calling `_on_complete_cb`. Per CONTEXT.md Area 1, max duration auto-stop is "treat as normal stop" and Area 2 defines normal stop as descending chime. Config is loaded the same way as `_stop_recording` to get `sounds_enabled` and `sounds_output_device`.
- Do NOT use suppress=True on the Listener — crashes X11 sessions (pynput issue #269).

def _enumerate_sinks(self):
    from linux_speech_flow.audio import list_sinks
    self._sinks = list_sinks()
    self._sink_combo.remove_all()
    # Prepend a "System default" entry (empty name = no --device flag to paplay)
    self._sink_combo.append_text("System default")
    current = self._config.get("sounds_output_device", "")
    selected_idx = 0  # default: "System default"
    for i, sink in enumerate(self._sinks):
        self._sink_combo.append_text(sink["description"])
        if sink["name"] == current:
            selected_idx = i + 1  # +1 because index 0 = "System default"
    self._sink_combo.set_active(selected_idx)
